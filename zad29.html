Teoria Programowania Obiektowego

1. W jaki sposób można określić właściwość obiektu?

Właściwość obiektu można określić na dwa sposoby:

    let osoba = { imie: "Jan", wiek: 30 };
    console.log(osoba.imie); // Jan
    console.log(osoba["wiek"]); // 30

2. W jaki sposób można odwołać się do pól i metod?

    let samochod = {
        marka: "Toyota",
        pokazMarke: function() {
            console.log(this.marka);
        }
    };
    samochod.pokazMarke(); // Toyota

3. Cechy charakterystyczne konstruktora

  • Jest specjalną funkcją do tworzenia obiektów.
  • Używa słowa kluczowego this.
  • Wywoływany z new.

4. Konstruowanie obiektu

    let telefon = {
        marka: "Samsung",
        model: "Galaxy S21",
        cena: 3000,
        wyswietlInformacje: function() {
            console.log(`Marka: ${this.marka}, Model: ${this.model}, Cena: ${this.cena} PLN`);
        }
    };
    telefon.wyswietlInformacje();

5. Znaczenie słowa this

    let osoba = {
        imie: "Anna",
        wiek: 25,
        przedstawSie: function() {
            console.log(`Cześć, mam na imię ${this.imie}`);
        }
    };
    osoba.przedstawSie();

6. Konstruktor w JS

    function Osoba(imie, wiek) {
        this.imie = imie;
        this.wiek = wiek;
        this.przedstawSie = function() {
            console.log(`Cześć, jestem ${this.imie} i mam ${this.wiek} lat.`);
        };
    }
    let osoba1 = new Osoba("Tomek", 28);
    osoba1.przedstawSie();

7. Znaczenie słowa kluczowego new

    function Samochod(marka, model) {
        this.marka = marka;
        this.model = model;
    }
    let auto1 = new Samochod("BMW", "X5");
    console.log(auto1.marka); // BMW

8. Znaczenie słowa kluczowego prototype

    function Osoba(imie, wiek) {
        this.imie = imie;
        this.wiek = wiek;
    }
    Osoba.prototype.przedstawSie = function() {
        console.log(`Cześć, jestem ${this.imie} i mam ${this.wiek} lat.`);
    };
    let osoba1 = new Osoba("Marta", 30);
    osoba1.przedstawSie();
